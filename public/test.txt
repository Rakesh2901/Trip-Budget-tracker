    <script>
        let token = localStorage.getItem('token');
        let currentTrip = null;
        let mainChart = null;

        // --- GLOBAL KEY LISTENER FOR "ENTER" KEY ---
        function handleEnter(event, callback) {
            if (event.key === 'Enter') {
                event.preventDefault(); // Stop default form submission if inside a form tag
                callback();
            }
        }

        // --- AUTH LOGIC ---
        if(token) initApp();

        function toggleAuthMode() {
            const loginForm = document.getElementById('login-form');
            const regForm = document.getElementById('register-form');
            const title = document.getElementById('auth-subtitle');
            
            if(loginForm.classList.contains('hidden')) {
                loginForm.classList.remove('hidden');
                regForm.classList.add('hidden');
                title.innerText = "Sign in to manage your travel budget";
            } else {
                loginForm.classList.add('hidden');
                regForm.classList.remove('hidden');
                title.innerText = "Create an account to get started";
            }
        }

        async function login() {
            const email = document.getElementById('login-email').value;
            const password = document.getElementById('login-pass').value;
            if(!email || !password) return showError('login', 'Please fill in all fields');

            const res = await fetch('/api/login', {
                method: 'POST', headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify({ email, password })
            });
            const data = await res.json();
            
            if(res.ok) {
                localStorage.setItem('token', data.token);
                token = data.token;
                initApp();
            } else {
                showError('login', data.msg);
            }
        }

        async function register() {
            const username = document.getElementById('reg-name').value;
            const email = document.getElementById('reg-email').value;
            const password = document.getElementById('reg-pass').value;
            if(!username || !email || !password) return showError('reg', 'Please fill in all fields');

            const res = await fetch('/api/register', {
                method: 'POST', headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify({ username, email, password })
            });
            const data = await res.json();
            
            if(res.ok) {
                localStorage.setItem('token', data.token);
                token = data.token;
                initApp();
            } else {
                showError('reg', data.msg);
            }
        }

        function logout() {
            localStorage.removeItem('token');
            location.reload();
        }

        function showError(type, msg) {
            document.getElementById(type + '-error').innerText = msg;
        }

        // --- APP LOGIC ---
        async function initApp() {
            document.getElementById('auth-screen').style.display = 'none';
            document.getElementById('app-container').classList.remove('hidden');

            const userRes = await fetch('/api/auth/user', { headers: { 'Authorization': token } });
            if(userRes.ok) {
                const user = await userRes.json();
                document.getElementById('user-name').innerText = user.username;
                document.getElementById('user-email').innerText = user.email;
            } else {
                logout();
            }
            loadData();
        }

        async function loadData() {
            const res = await fetch('/api/trips', { headers: { 'Authorization': token } });
            const trips = await res.json();
            
            if(trips.length > 0) {
                currentTrip = trips[trips.length - 1];
                renderDashboard(currentTrip);
            }
        }

        function renderDashboard(trip) {
            const total = trip.budget;
            const spent = trip.expenses.reduce((a, b) => a + b.amount, 0);
            const remaining = total - spent;
            
            document.getElementById('total-budget').innerText = total.toLocaleString(undefined, {style:'currency', currency:'USD'});
            document.getElementById('total-expenses').innerText = spent.toLocaleString(undefined, {style:'currency', currency:'USD'});
            document.getElementById('total-remaining').innerText = remaining.toLocaleString(undefined, {style:'currency', currency:'USD'});

            const today = new Date();
            const start = new Date(trip.startDate);
            const days = Math.ceil((start - today) / (1000 * 60 * 60 * 24));
            document.getElementById('days-count').innerText = days > 0 ? days : 0;

            initChart(trip);
        }

        function switchTab(view) {
            document.querySelectorAll('.main > div').forEach(d => d.classList.add('hidden'));
            document.getElementById('view-' + view).classList.remove('hidden');
            
            document.querySelectorAll('.menu li').forEach(li => li.classList.remove('active'));
            event.currentTarget.classList.add('active');
        }

        async function createTrip() {
            const destination = document.getElementById('trip-dest').value;
            const budget = document.getElementById('trip-budget').value;
            const startDate = document.getElementById('trip-date').value;
            
            await fetch('/api/trips', {
                method: 'POST',
                headers: { 'Content-Type': 'application/json', 'Authorization': token },
                body: JSON.stringify({ destination, budget, startDate })
            });
            alert("Trip Created Successfully");
            loadData();
        }

        async function addExpense() {
            if(!currentTrip) return alert("Please create a trip in Settings first.");
            const desc = document.getElementById('exp-desc').value;
            const amount = document.getElementById('exp-amount').value;
            const category = document.getElementById('exp-cat').value;

            if(!desc || !amount) return alert("Please fill in description and amount");

            await fetch(`/api/trips/${currentTrip._id}/expenses`, {
                method: 'POST',
                headers: { 'Content-Type': 'application/json', 'Authorization': token },
                body: JSON.stringify({ description: desc, amount, category })
            });
            
            // Clear inputs
            document.getElementById('exp-desc').value = '';
            document.getElementById('exp-amount').value = '';
            
            alert("Expense Added");
            loadData();
        }

        function initChart(trip) {
            const ctx = document.getElementById('mainChart').getContext('2d');
            if(mainChart) mainChart.destroy();

            const cats = ['Lodging', 'Transportation', 'Food', 'Entertainment', 'Other'];
            const data = cats.map(c => trip.expenses.filter(e => e.category === c).reduce((sum, e) => sum + e.amount, 0));

            mainChart = new Chart(ctx, {
                type: 'bar',
                data: {
                    labels: cats,
                    datasets: [{
                        label: 'Amount Spent',
                        data: data,
                        backgroundColor: '#38bdf8',
                        borderRadius: 4
                    }]
                },
                options: {
                    responsive: true,
                    maintainAspectRatio: false,
                    plugins: { legend: { display: false } },
                    scales: {
                        x: { grid: { display: false }, ticks: { color: '#94a3b8' } },
                        y: { grid: { color: 'rgba(255,255,255,0.05)' }, ticks: { color: '#94a3b8' } }
                    }
                }
            });
        }
    </script>
